


// Printing Last Digit 

#include<stdio.h>

int main()
{
    int num1 , mod;
    
    printf("Enter A Number ");
    scanf("%d",&num1);
    
    mod = num1%10;
    
    printf("Last Digit Is %d",mod);
    
    
    return 0;
}
