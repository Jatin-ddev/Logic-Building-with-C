#include<stdio.h>

int main()
{

float num1 , num2 , num3 , avg;

printf(" Enter 3 Numbers : ");
scanf("%f %f %f",&num1,&num2,&num3);

avg = (num1+num2+num3)/3;

printf("The Average is %f",avg);

return 0;
}
